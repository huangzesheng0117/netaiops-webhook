#
# Usage:
#   bash tools/v11_governance_master_runner.txt prepare 0
#   bash tools/v11_governance_master_runner.txt status 0
#   bash tools/v11_governance_master_runner.txt verify 0
#   CONFIRM_COMMIT=YES bash tools/v11_governance_master_runner.txt commit 0
#   CONFIRM_COMMIT=YES CONFIRM_PUSH=YES bash tools/v11_governance_master_runner.txt commit 0
#   CONFIRM_ROLLBACK=YES bash tools/v11_governance_master_runner.txt rollback 0
#
# Safety:
# - The runner does not generate business implementation code.
# - The runner does not call real GLM, Prometheus MCP, Netmiko MCP, or DingTalk.
# - The runner never pushes unless CONFIRM_PUSH=YES.
# - prepare requires a clean Git worktree and /health status=ok.
# - rollback is strictly pre-commit. If HEAD changed, use git revert.
# - No git reset --hard, git clean -fd, git add ., or git add -A.

${GLM52_REQUEST_ID}
${FULL_SUCCESS_REQUEST_ID}}"

Required confirmations:
  CONFIRM_COMMIT=YES
  CONFIRM_PUSH=YES       # optional; otherwise push is skipped
  CONFIRM_ROLLBACK=YES
  CONFIRM_REPREPARE=YES  # only when replacing stale /tmp state
EOF
}

  printf '[%s] %s\n' "$(date '+%F %T')" "$*"

fail() {
  log "FAIL: $*"
  exit 1

batch_files_array() {
  mapfile -t CURRENT_FILES < <(
    printf '%s\n' "${FILES[${BATCH}]}" |
      sed '/^[[:space:]]*$/d'
  )
}

fixture_ids_array() {
  mapfile -t CURRENT_FIXTURE_IDS < <(
    printf '%s\n' "${REAL_FIXTURE_IDS}" |
      sed '/^[[:space:]]*$/d'
  )
}

branch_check() {
  local branch
  branch="$(git rev-parse --abbrev-ref HEAD)"
  [[ "${branch}" == "${EXPECTED_BRANCH}" ]] ||
    fail "expected branch ${EXPECTED_BRANCH}, got ${branch}"
}

current_head() {
  git rev-parse HEAD
}

changed_paths() {
  {
    git diff --name-only
    git diff --cached --name-only
    git ls-files --others --exclude-standard
  } | sed '/^[[:space:]]*$/d' | LC_ALL=C sort -u
}

clean_worktree_check() {
  local status
  status="$(git status --short --untracked-files=all)"
  if [[ -n "${status}" ]]; then
    echo "${status}"
    fail "worktree must be clean before prepare"
  fi
}

health_status_check() {
  local output_file="${1}"
  mkdir -p "$(dirname "${output_file}")"
  curl -fsS --max-time 5 "${BASE_URL}/health" > "${output_file}"
  python - "${output_file}" <<'PY'
import json
import sys
from pathlib import Path

path = Path(sys.argv[1])
data = json.loads(path.read_text(encoding="utf-8"))
status = str(data.get("status", "")).strip().lower()
if status != "ok":
    raise SystemExit(f"/health status must be ok, got {status!r}")
print(json.dumps(data, ensure_ascii=False, indent=2))
PY
}

wait_health() {
  local i
  mkdir -p "${STATE_DIR}"
  for i in $(seq 1 45); do
    if curl -fsS --max-time 5 "${BASE_URL}/health" > "${HEALTH_FILE}" 2>/dev/null; then
      if python - "${HEALTH_FILE}" <<'PY'
import json
import sys
from pathlib import Path

data = json.loads(Path(sys.argv[1]).read_text(encoding="utf-8"))
raise SystemExit(0 if str(data.get("status", "")).lower() == "ok" else 1)
PY
      then
        log "service ready at attempt ${i}"
        python -m json.tool "${HEALTH_FILE}"
        return 0
      fi
    fi
    sleep 2
  done

  sudo systemctl status "${SERVICE_NAME}" --no-pager || true
  sudo journalctl -u "${SERVICE_NAME}" -n 160 --no-pager || true
  return 1
}

baseline_head_check() {
  [[ -f "${HEAD_FILE}" ]] ||
    fail "baseline HEAD is missing; run prepare first"

  local baseline now
  baseline="$(cat "${HEAD_FILE}")"
  now="$(current_head)"
  [[ "${now}" == "${baseline}" ]] ||
    fail "HEAD changed after prepare: baseline=${baseline}, current=${now}; rollback is not allowed, use git revert for committed changes"
}

  echo "branch=$(git rev-parse --abbrev-ref HEAD)"

}

prepare_batch() {
  branch_check
  clean_worktree_check

  local health_tmp
  health_tmp="$(mktemp)"
  trap 'rm -f "${health_tmp:-}"' RETURN
  health_status_check "${health_tmp}" >/dev/null

  if [[ -e "${STATE_DIR}" ]]; then
    if [[ "${CONFIRM_REPREPARE:-}" != "YES" ]]; then
      fail "state already exists: ${STATE_DIR}; set CONFIRM_REPREPARE=YES only after confirming it is stale"
    fi
    rm -rf "${STATE_DIR}"
  fi

  batch_files_array
  mkdir -p "${BACKUP_DIR}"
  current_head > "${HEAD_FILE}"

  python - \
    "${MANIFEST}" \
    "${PROJECT_ROOT}" \
    "${BACKUP_DIR}" \
    "$(git rev-parse --abbrev-ref HEAD)" \
    "$(cat "${HEAD_FILE}")" \
    "${CURRENT_FILES[@]}" <<'PY'
import hashlib
import json
import shutil
import sys
from datetime import datetime, timezone
from pathlib import Path

manifest_path = Path(sys.argv[1])
root = Path(sys.argv[2])
backup = Path(sys.argv[3])
branch = sys.argv[4]
head = sys.argv[5]
files = sys.argv[6:]

items = []
for rel in files:
    path = root / rel
    item = {
        "path": rel,
        "existed": path.exists(),
        "is_file": path.is_file(),
        "is_symlink": path.is_symlink(),
        "sha256": "",
    }
    if path.is_symlink():
        raise SystemExit(f"planned path must not be a symlink: {rel}")
    if path.is_file():
        data = path.read_bytes()
        item["sha256"] = hashlib.sha256(data).hexdigest()
        target = backup / rel
        target.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(path, target)
    items.append(item)

payload = {
    "schema_version": 2,
    "generated_at": datetime.now(timezone.utc).isoformat(),
    "branch": branch,
    "head": head,
    "files": items,
}
manifest_path.parent.mkdir(parents=True, exist_ok=True)
manifest_path.write_text(
    json.dumps(payload, ensure_ascii=False, indent=2) + "\n",
    encoding="utf-8",
)
print(manifest_path)
PY

  rm -f "${VERIFY_STAMP}"
  log "prepared batch ${BATCH}"
  log "baseline head: $(cat "${HEAD_FILE}")"
  log "manifest: ${MANIFEST}"
}

compile_files() {
  batch_files_array
  local py_files=()
  local file

  for file in "${CURRENT_FILES[@]}"; do
    [[ -e "${file}" ]] || fail "planned file missing: ${file}"
    [[ ! -L "${file}" ]] || fail "planned file must not be a symlink: ${file}"
    if [[ "${file}" == *.py ]]; then
      py_files+=("${file}")
    fi
  done

  if ((${#py_files[@]})); then
    python -m py_compile "${py_files[@]}"
  fi
}

run_batch_tests() {
  python -m unittest discover \
    -s tests \
    -p "${TEST_PATTERN[${BATCH}]}" \
    -v
}

verify_git_scope() {
  batch_files_array

  local actual expected
  actual="$(changed_paths)"
  expected="$(printf '%s\n' "${CURRENT_FILES[@]}" | LC_ALL=C sort -u)"

  if [[ "${actual}" != "${expected}" ]]; then
    echo "[EXPECTED]"
    printf '%s\n' "${expected}"
    echo "[ACTUAL]"
    printf '%s\n' "${actual}"
    fail "worktree scope does not match planned batch files"
  fi

  git diff --check
  git diff --cached --check
}

forbidden_file_check() {
  local files
  files="$(changed_paths)"

  if printf '%s\n' "${files}" |
    egrep -q '(^|/)(config\.yaml|data|logs|backup|venv|__pycache__)(/|$)|(^|/)(\.env|[^/]+\.env)$|\.(zip|tar|tgz|gz|bz2|xz|7z)$'; then
    printf '%s\n' "${files}"
    fail "forbidden runtime, secret, environment, or archive path detected"
  fi
}

compute_verification_digest() {
  batch_files_array
  python - "$(cat "${HEAD_FILE}")" "${PROJECT_ROOT}" "${CURRENT_FILES[@]}" <<'PY'
import hashlib
import sys
from pathlib import Path

head = sys.argv[1]
root = Path(sys.argv[2])
files = sys.argv[3:]

digest = hashlib.sha256()
digest.update(head.encode("utf-8"))
digest.update(b"\0")

for rel in sorted(files):
    path = root / rel
    if not path.is_file():
        raise SystemExit(f"missing file for verification digest: {rel}")
    digest.update(rel.encode("utf-8"))
    digest.update(b"\0")
    digest.update(hashlib.sha256(path.read_bytes()).digest())
    digest.update(b"\0")

print(digest.hexdigest())
PY
}

write_verification_stamp() {
  compute_verification_digest > "${VERIFY_STAMP}"
}

verification_stamp_matches() {
  [[ -f "${VERIFY_STAMP}" ]] || return 1
  local expected actual
  expected="$(cat "${VERIFY_STAMP}")"
  actual="$(compute_verification_digest)"
  [[ "${expected}" == "${actual}" ]]
}

verify_batch() {
  branch_check
  [[ -f "${MANIFEST}" ]] ||
    fail "prepare state missing; run prepare first"
  baseline_head_check

  compile_files
  run_batch_tests
  verify_git_scope
  forbidden_file_check
  extra_verify
  write_verification_stamp

  log "batch ${BATCH} verification passed"
  log "verification digest: $(cat "${VERIFY_STAMP}")"
}

commit_batch() {
  [[ "${CONFIRM_COMMIT:-}" == "YES" ]] ||
    fail "set CONFIRM_COMMIT=YES"

  branch_check
  [[ -f "${MANIFEST}" ]] ||
    fail "prepare state missing; run prepare first"
  baseline_head_check

  if verification_stamp_matches; then
    log "reusing valid verification stamp; full verify is not repeated"
  else
    log "verification stamp missing or stale; running full verify"
    verify_batch
  fi

  batch_files_array
  git add -- "${CURRENT_FILES[@]}"

  git diff --cached --name-only
  git diff --cached --stat
  git diff --cached --check

  local staged expected
  staged="$(git diff --cached --name-only | LC_ALL=C sort -u)"
  expected="$(printf '%s\n' "${CURRENT_FILES[@]}" | LC_ALL=C sort -u)"

  [[ "${staged}" == "${expected}" ]] ||
    fail "staged scope differs from planned scope"

  [[ -z "$(git diff --name-only)" ]] ||
    fail "unstaged changes remain after precise staging"

  git commit -m "${COMMIT_MESSAGE[${BATCH}]}"

  if [[ "${CONFIRM_PUSH:-}" == "YES" ]]; then
    git push origin "${EXPECTED_BRANCH}"
  else
    log "push skipped; set CONFIRM_PUSH=YES to push"
  fi

  git log --oneline -5
  git status --short --untracked-files=all
}

rollback_batch() {
  [[ "${CONFIRM_ROLLBACK:-}" == "YES" ]] ||
    fail "set CONFIRM_ROLLBACK=YES"

  branch_check
  [[ -f "${MANIFEST}" ]] ||
    fail "manifest missing; cannot perform safe rollback"
  baseline_head_check

  batch_files_array
  git restore --staged -- "${CURRENT_FILES[@]}" 2>/dev/null || true

  python - "${MANIFEST}" "${PROJECT_ROOT}" "${BACKUP_DIR}" <<'PY'
import json
import shutil
import sys
from pathlib import Path

manifest = json.loads(Path(sys.argv[1]).read_text(encoding="utf-8"))
root = Path(sys.argv[2])
backup = Path(sys.argv[3])

for item in manifest["files"]:
    rel = item["path"]
    target = root / rel
    source = backup / rel

    if item["existed"]:
        if not source.is_file():
            raise SystemExit(f"backup missing: {source}")
        target.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(source, target)
        print("restored", rel)
    else:
        if target.is_file() or target.is_symlink():
            target.unlink()
            print("removed", rel)
PY

  rm -f "${VERIFY_STAMP}"

  local remaining
  remaining="$(changed_paths)"
  if [[ -n "${remaining}" ]]; then
    echo "${remaining}"
    fail "rollback left unexpected worktree changes"
  fi

  if [[ "${RESTART_REQUIRED[${BATCH}]}" == "yes" ]]; then
    sudo systemctl restart "${SERVICE_NAME}"
    wait_health
  fi

  git status --short --untracked-files=all
  log "batch ${BATCH} pre-commit rollback completed"
}

case "${ACTION}" in

    usage

    ;;
esac
